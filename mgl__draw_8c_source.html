451 void mgl_draw_bezier_to_surface(SDL_Surface *surface, MglVec2D p0, MglVec2D p1, MglVec2D p2, MglVec4D color)
452 {
453  MglVec2D qp,qp2,qpv; /*approximation line starting point and vector*/
454  MglVec2D p0v,p1v,temp; /*vectors from point to next point*/
455  MglVec2D dp; /*draw point*/
456  MglFloat t = 0; /*time segment*/
457  MglFloat tstep;
458  MglFloat totalLength;
459  totalLength = mgl_vec2d_magnitude(p0)+mgl_vec2d_magnitude(p1)+mgl_vec2d_magnitude(p2);
460  if (totalLength == 0)
461  {
462  mgl_logger_warn("path provided is zero length.");
463  return;
464  }
465  tstep = fabs(1.0/(totalLength * 0.9));
466  mgl_vec2d_sub(p0v,p1,p0);
467  mgl_vec2d_sub(p1v,p2,p1);
468  for (t = 0; t <= 1;t += tstep)
469  {
470  /*calculate Q*/
471  mgl_vec2d_scale(temp,p0v,t);
472  mgl_vec2d_add(qp,p0,temp);
473 
474  mgl_vec2d_scale(temp,p1v,t);
475  mgl_vec2d_add(qp2,p1,temp);
476 
477  mgl_vec2d_sub(qpv,qp2,qp);
478 
479  mgl_vec2d_scale(temp,qpv,t);
480  mgl_vec2d_add(dp,qp,temp);
481 
482  mgl_draw_pixel_to_surface(surface,dp,color);
483  }
