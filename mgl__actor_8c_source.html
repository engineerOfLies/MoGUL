292 MglBool mgl_actor_next_frame(MglActor * actor)
293 {
294  MglAction *action;
295  MglCallback *cb;
296  MglUint now,diff;
297  MglFloat timefactor;
298  if (!actor)return MglFalse;
299  if (!actor->action)return MglFalse;
300  action = actor->action;
301  actor->lastFrame = actor->frame;
302 
303  if ((!__mgl_actor_use_timing)||(__mgl_ticks_per_frame == 0))
304  {
305  /*driven by calls to this function*/
306  actor->frame = actor->frame + (actor->frameRate * action->frameRate * actor->frameDirection);
307  }
308  else
309  {
310  /*based on sdl_ticks*/
311  now = mgl_graphics_get_render_time();
312  diff = now - actor->lastTime;
313  timefactor = (MglFloat)diff / __mgl_ticks_per_frame;
314  actor->frame = actor->frame + (actor->frameRate * action->frameRate * actor->frameDirection)*timefactor;
315  actor->lastTime = now;
316  }
317  if (actor->frame > action->end)
318  {
319  switch(actor->action->type)
320  {
321  case MglActionLoop:
322  actor->frame = actor->action->begin;
323  return MglFalse;
324  case MglActionPass:
325  actor->frame = actor->action->end;
326  return MglTrue;
327  case MglActionOsci:
328  actor->frame = actor->action->end;
329  actor->frameDirection = -1;
330  return MglFalse;
331  }
332  if (action->onEnd.function != NULL)
333  {
334  action->onEnd.function(action->onEnd.data);
335  }
336  }
337  if (actor->frame < actor->action->begin)
338  {
339  switch(actor->action->type)
340  {
341  case MglActionLoop:
342  actor->frame = actor->action->begin;
343  return MglFalse;
344  case MglActionPass:
345  actor->frame = actor->action->begin;
346  return MglFalse;
347  case MglActionOsci:
348  actor->frame = actor->action->begin;
349  actor->frameDirection = 1;
350  return MglTrue;
351  }
352  if (action->onBegin.function != NULL)
353  {
354  action->onBegin.function(action->onBegin.data);
355  }
356  }
357  if ((MglUint)actor->lastFrame != (MglUint)actor->frame)
358  {
359  cb = g_hash_table_lookup(action->callbacks,GINT_TO_POINTER((MglUint)actor->frame));
360  if (cb != NULL)
361  {
362  if (cb->function)
363  {
364  cb->function(cb->data);
365  }
366  }
367  }
